package main

import (
	"fmt"
	"os"
)

func main() {
	// Retrieve the command line argument
	if len(os.Args) < 2 {
		fmt.Println("Usage: ./out <param>")
		return
	}
	param := os.Args[1]

	// Random string placeholder
	secretCode := "PLACEHOLDER_RANDOM_STRING"

	// Print the random string and the command line parameter
	fmt.Printf("Hello %s! The secret code is %s.", param, secretCode)
}
